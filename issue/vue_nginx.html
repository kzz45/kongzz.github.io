          # Vue项目部署到NGINX

          # 前端Vue资源打包

          打包后的文件存在Vue项目目录下的dist目录

          npm run build:prod

          # 部署到nginx root目录

          将dist目录下的文件上传到nginx服务器的root目录下(nginx 配置文件的中的root配置)

          # nginx 配置文件

          user nginx;
worker_processes auto;
pid /run/nginx.pid;
error_log /var/log/nginx/error.log;
include /usr/share/nginx/modules/*.conf;
events {
    worker_connections 1024;
}
http {
    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';
    access_log          /var/log/nginx/access.log  main;
    sendfile            on;
    tcp_nopush          on;
    tcp_nodelay         on;
    keepalive_timeout   65;
    types_hash_max_size 4096;
    include             /etc/nginx/mime.types;
    default_type        application/octet-stream;
    include             /etc/nginx/conf.d/*.conf;
    server {
        listen       80;
        listen       [::]:80;
        server_name  your-server-name.com;
        root         /data/dist;
	location / {
		root         /data/dist;
		index        index.html index.htm;
		try_files    $uri $uri/ /index.html; # 为了防止Vue 路由刷新的时候出现404
	}
    # /prod-api/ 是你Vue项目中VUE_APP_BASE_API的配置
    location /prod-api/ {
       proxy_set_header        X-Real-IP        $remote_addr;
       proxy_set_header        X-Forwarded-For  $proxy_add_x_forwarded_for;
       proxy_set_header        Host             $http_host;
       proxy_set_header        X-NginX-Proxy    true;
       proxy_set_header        Connection "";
       add_header              Access-Control-Allow-Origin *;
       proxy_connect_timeout   300;
       proxy_send_timeout      300;
       proxy_read_timeout      300;
       proxy_pass              http://your-backend-server:8000/;
    }
    
    error_page 404 /404.html;
    location = /404.html {
    }

    error_page 500 502 503 504 /50x.html;
    location = /50x.html {
    }
    }
}

          # 启动nginx服务

          systemctl start nginx
